14def sizeup():
15    """Print the sub-directories and their sizes of the current working directory."""
16    parser = argparse.ArgumentParser("sizeup")
17    parser.add_argument(
18        "-i",
19        "--ignore",
20        nargs="*",
21        default=[],
22        type=str,
23        help="Directory patterns to ignore.",
24    )
25    args = parser.parse_args()
26    matcher = younotyou.Matcher(exclude_patterns=args.ignore)
27    sizes: dict[str, int] = {}
28    folders = [
29        folder
30        for folder in Pathier.cwd().iterdir()
31        if folder.is_dir() and str(folder) in matcher
32    ]
33    print(f"Sizing up {len(folders)} directories...")
34    for folder in printbuddies.track(folders, "Scanning directories"):
35        try:
36            sizes[folder.name] = folder.size
37        except Exception as e:
38            pass
39    total_size = sum(sizes[folder] for folder in sizes)
40    size_list = [
41        (folder, Pathier.format_bytes(sizes[folder]))
42        for folder in sorted(list(sizes.keys()), key=lambda f: sizes[f], reverse=True)
43    ]
44    print(griddle.griddy(size_list, ["Dir", "Size"]))
45    print(f"Total size of '{Pathier.cwd()}': {Pathier.format_bytes(total_size)}")

108    def modified_since_last_read(self) -> bool:
109        """Returns `True` if this file hasn't been read from or has been modified since the last time this object
110        made a call to `self.read_text()`, `self.read_bytes()`, or `self.open(mode="r"|"rb")`.
111
112        Note: This property is only relative to the lifetime of this `Pathier` instance, not the file itself.

128    def size(self) -> int:
129        """Returns the size in bytes of this file or directory.
130
131        If this path doesn't exist, `0` will be returned."""
132        if not self.exists():
133            return 0
134        elif self.is_file():
135            return self.stat().st_size
136        elif self.is_dir():

145    @staticmethod
146    def format_bytes(size: int) -> str:
147        """Format `size` with common file size abbreviations and rounded to two decimal places.
148        >>> 1234 -> "1.23 kb" """
149        unit = "bytes"
150        for unit in ["bytes", "kb", "mb", "gb", "tb", "pb"]:
151            if unit != "bytes":
152                size *= 0.001  # type: ignore
153            if size < 1000 or unit == "pb":
154                break
155        return f"{round(size, 2)} {unit}"

206    def _get_path_part_rindex(self, name: str) -> int | None:
207        """Returns the index value for the last occurrence of `name` if it is in this path instance."""
208        if name not in self.parts:
209            raise Exception(f"{name} is not a parent of {self}")
210        lastdex = len(self.parts) - 1
211        rev_parts = self.parts[::-1]
212        return lastdex - rev_parts.index(name)

214    def moveup(self, name: str) -> Self:
215        """Return a new `Pathier` object that is a parent of this instance.
216
217        `name` is case-sensitive and raises an exception if it isn't in `self.parts`.
218        >>> p = Pathier("C:/some/directory/in/your/system")
219        >>> print(p.moveup("directory"))
220        >>> "C:/some/directory"
221        >>> print(p.moveup("yeet"))
222        >>> "Exception: yeet is not a parent of C:/some/directory/in/your/system" """
223        index = self._get_path_part_rindex(name)
224        if index is None:
225            raise Exception(f"{name} is not a parent of {self}")
226        return self - (len(self.parts) - index - 1)
227        return self.__class__(*(self.parts[: index + 1]))
228
229    def __sub__(self, levels: int) -> Self:
230        """Return a new `Pathier` object moved up `levels` number of parents from the current path.
231        >>> p = Pathier("C:/some/directory/in/your/system")
232        >>> new_p = p - 3
233        >>> print(new_p)
234        >>> "C:/some/directory" """
235        path = self
236        for _ in range(levels):
237            path = path.parent
238        return path

240    def move_under(self, name: str) -> Self:
241        """Return a new `Pathier` object such that the stem is one level below the given folder `name`.
242
243        `name` is case-sensitive and raises an exception if it isn't in `self.parts`.
244        >>> p = Pathier("a/b/c/d/e/f/g")
245        >>> print(p.move_under("c"))
246        >>> 'a/b/c/d'"""
247        index = self._get_path_part_rindex(name)
248        if index is None:
249            raise Exception(f"{name} is not a parent of {self}")
250        return self - (len(self.parts) - index - 2)

252    def separate(self, name: str, keep_name: bool = False) -> Self:
253        """Return a new `Pathier` object that is the relative child path after `name`.
254
255        `name` is case-sensitive and raises an exception if it isn't in `self.parts`.
256
257        #### :params:
258
259        `keep_name`: If `True`, the returned path will start with `name`.
260        >>> p = Pathier("a/b/c/d/e/f/g")
261        >>> print(p.separate("c"))
262        >>> 'd/e/f/g'
263        >>> print(p.separate("c", True))
264        >>> 'c/d/e/f/g'"""
265        if name not in self.parts:
266            raise Exception(f"{name} is not a parent of {self}")
267        if keep_name:
268            return self.__class__(*self.parts[self.parts.index(name) :])
269        return self.__class__(*self.parts[self.parts.index(name) + 1 :])

301    def write_text(
302        self,
303        data: Any,
304        encoding: Any | None = None,
305        errors: Any | None = None,
306        newline: Any | None = None,
307        parents: bool = True,
308    ) -> int:
309        """Write data to file.
310
311        If a `TypeError` is raised, the function  will attempt to cast `data` to a `str` and try the write again.
312
313        If a `FileNotFoundError` is raised and `parents = True`, `self.parent` will be created.
314        """
315        write = functools.partial(
316            super().write_text,
317            encoding=encoding,
318            errors=errors,
319            newline=newline,
320        )
321        try:
322            return write(data)
323        except TypeError:
324            data = str(data)
325            return write(data)
326        except FileNotFoundError:
327            if parents:
328                self.parent.mkdir(parents=True)
329                return write(data)
330            else:
331                raise
332        except Exception as e:
333            raise

453    def toml_dumps(
454        self,
455        data: Any,
456        toml_encoders: Sequence[Callable[[Any], Any]] = [str],
457        encoding: Any | None = None,
458        errors: Any | None = None,
459        newline: Any | None = None,
460        sort_keys: bool = False,
461        parents: bool = True,
462    ):
463        """Dump `data` to toml file.
464
465        `toml_encoders` can be a list of functions to call when a value in `data` doesn't map to `tomlkit`'s built in types.
466        By default, anything that `tomlkit` can't convert will be cast to a string. Encoder order matters.
467        e.g. By default any `Pathier` object in `data` will be converted to a string."""
468        encoders: list[Callable[[Any], Any]] = []
469        for toml_encoder in toml_encoders:
470            encoder: Callable[[Any], Any] = lambda x: tomlkit.item(  # type:ignore
471                toml_encoder(x)
472            )
473            encoders.append(encoder)
474            tomlkit.register_encoder(encoder)
475        try:
476            self.write_text(
477                tomlkit.dumps(data, sort_keys),  # type:ignore
478                encoding,
479                errors,
480                newline,
481                parents,
482            )
483        except Exception as e:
484            raise e
485        finally:
486            for encoder in encoders:
487                tomlkit.unregister_encoder(encoder)

546    def copy(
547        self, new_path: Self | pathlib.Path | str, overwrite: bool = False
548    ) -> Self:
549        """Copy the path pointed to by this instance
550        to the instance pointed to by `new_path` using `shutil.copyfile`
551        or `shutil.copytree`.
552
553        Returns the new path.
554
555        #### :params:
556
557        `new_path`: The copy destination.
558
559        `overwrite`: If `True`, files already existing in `new_path` will be overwritten.
560        If `False`, only files that don't exist in `new_path` will be copied."""
561        dst = self.__class__(new_path)
562        if self.is_dir():
563            if overwrite or not dst.exists():
564                dst.mkdir()
565                shutil.copytree(self, dst, dirs_exist_ok=True)
566            else:
567                files = self.rglob("*.*")
568                for file in files:
569                    dst = dst.with_name(file.name)
570                    if not dst.exists():
571                        shutil.copyfile(file, dst)
572        elif self.is_file():
573            if overwrite or not dst.exists():
574                shutil.copyfile(self, dst)
575        return dst

577    def backup(self, timestamp: bool = False) -> Self | None:
578        """Create a copy of this file or directory with `_backup` appended to the path stem.
579        If the path to be backed up doesn't exist, `None` is returned.
580        Otherwise a `Pathier` object for the backup is returned.
581
582        #### :params:
583
584        `timestamp`: Add a timestamp to the backup name to prevent overriding previous backups.
585
586        >>> path = Pathier("some_file.txt")
587        >>> path.backup()
588        >>> list(path.iterdir())
589        >>> ['some_file.txt', 'some_file_backup.txt']
590        >>> path.backup(True)
591        >>> list(path.iterdir())
592        >>> ['some_file.txt', 'some_file_backup.txt', 'some_file_backup_04-28-2023-06_25_52_PM.txt']
593        """
594        if not self.exists():
595            return None
